\newcommand{\makeTitlePage}{%
  \thispagestyle{empty}
  \mbox{}\vfill
  \begin{center}
    \def\cornerOrn{\pgfornament[width=1cm]{35}}%
    \def\vLineOrnNum{84}%
    \def\vLineOrnOffset{2.5mm}%
    \def\hLineOrnNum{82}%
    \def\hLineOrnOffset{-2.5mm}%
    \def\yOffset{20pt}%
    \begin{tikzpicture}[%
        every node/.style={%
          inner sep=0,
          node distance=0,
        },
      ]

      \node [font=\fontsize{36}{36}\selectfont\scshape\centering] (title) {%
        \parbox{0.71\textwidth}{%
          \centering\thisTitle%
        }};
      \node [below left=of title.north west, yshift=\yOffset] (UL) {\cornerOrn};
      \node [below right=of title.north east, yshift=\yOffset] (UR) {%
        \adjustbox{reflect}{\cornerOrn}%
      };
      \node [above left=of title.south west, yshift=-\yOffset] (LL) {%
        \adjustbox{rotate=90}{\cornerOrn}%
      };
      \node [above right=of title.south east, yshift=-\yOffset] (LR) {%
        \adjustbox{rotate=90,reflect}{\cornerOrn}%
      };
      \draw [transform canvas={xshift=\vLineOrnOffset}]
        (UL.south west) to [ornament=\vLineOrnNum] (LL.north west);
      \draw [transform canvas={xshift=-\vLineOrnOffset}]
        (UR.south east) to [ornament=\vLineOrnNum] (LR.north east);
      \draw [transform canvas={yshift=\hLineOrnOffset}]
        (UL.north east) to [ornament=\hLineOrnNum] (UR.north west);
      \draw [transform canvas={yshift=-\hLineOrnOffset}]
        (LL.south east) to [ornament=\hLineOrnNum] (LR.south west);
    \end{tikzpicture}

    \vspace{1.5\baselineskip}

    \noindent{\huge Version \thisSubtitle}
  \end{center}
  \vfill
  \mbox{}
  \vfill
  \mbox{}
  \newpage%
}
